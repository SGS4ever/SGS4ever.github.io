// LeetCode 78
class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<int> tmp;
        vector<vector<int>> res;

        res.push_back(tmp);  // empty set

        for (auto n : nums)
        {
            int sze = res.size();
            for (int i = 0; i < sze; i++)
            {
                tmp = res[i];
                tmp.push_back(n);
                res.push_back(tmp);
            }
        }

        return res;
    }
};

class Solution {
public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        vector<vector<int>> res;
        vector<int> tmp;
        res.push_back(tmp);
        int start, sze;

        sort(nums.begin(), nums.end());

        for (int idx = 0; idx < nums.size(); idx++)
        {
            if (idx > 0 && nums[idx] == nums[idx - 1])
            {
                start = sze;
            }
            else
            {
                start = 0;
            }
            sze = res.size();
            for (int i = start; i < sze; i++)
            {
                tmp = res[i];
                tmp.push_back(nums[idx]);
                res.push_back(tmp);
            }
        }

        return res;
    }
};
