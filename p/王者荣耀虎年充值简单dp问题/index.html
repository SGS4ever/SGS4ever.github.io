void maxPointsByMaxCost(int maxCost)
{
    int j = 0;
    vector<int> dp(maxCost + 1, 0);
    vector<int> lastPurchase(maxCost + 1, 0);

    {

            m++;
        }
    }
    
    printResultByMaxCost(dp, lastPurchase, maxCost);
    
    return ;
}
